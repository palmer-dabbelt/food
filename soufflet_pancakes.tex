\begin{recipe}{Soufflet Pancakes}
  \servings{2}

  \begin{ingredients}
    \ingredient{2}{large}{eggs}
    \ingredient{1}{tbsp}{flour}
    \ingredient{1}{tbsp}{sugar}
    \ingredient{}{}{vanilla extract}
    \ingredient{}{}{salt}
    \ingredient{}{}{butter}
  \end{ingredients}

  \begin{images}
    \begin{image}
      \includegraphics[width=\linewidth,trim=100px 100px 100px 100px, clip=true]{soufflet_pancakes-01.jpeg}
    \end{image}
  \end{images}

  \begin{steps}
  \item Cream the sugar and eggs yolks
  \item Add the salt, vanilla, and flour to the egg yolks
  \item Beat the egg whites to hard peaks
  \item Fold together the egg whites and yolk mixture
  \item Fry in butter until golden
  \end{steps}
\end{recipe}
